macro_rules! def_color {
    ($( $Id:ident => $str_lit:expr, )*) => {
        #[derive(Clone, PartialEq, Eq, PartialOrd, Ord, Debug)]
        pub enum Color { $($Id,)*
                     Custom(String),
        }
        impl Color {
            pub fn into_string(self) -> String {
                match self {
                    $(
                        Color::$Id => $str_lit.to_string(),
                    )*
                        Color::Custom(s) => s,
                }
            }
        }
    }
}
